# Pin

```js
import {
  Pin,
  PinMedia,
  PinTitle,
  PinNote,
  PinTags,
  PinActions,
} from 'piny-forest'
```

Represents a summary of saved link: visual representation (screenshot or other media),
title, note and associated tags.

Contains two variants `horizontal` (default) and `vertical`.
Without `PinMedia` both variants look the same .

```jsx live
<Pin
// variant="horizontal"
// variant="vertical"
>
  <PinMedia>
    <Link href="#image">
      <Image
        src="https://placekitten.com/2047/1536"
        pss={{
          size: '100%',
          objectFit: 'cover',
          boxShadow: 'black.20',
        }}
        alt="Cute kitten"
      />
    </Link>
  </PinMedia>
  <PinTitle>
    <Link href="#text">
      <Text as="h3">Heading</Text>
    </Link>
  </PinTitle>
  <PinNote>
    <Text as="p">Lorem ipsum dolor sit amet, consectetur adipiscing elit.</Text>
  </PinNote>
  <PinTags>
    <Tag href="#tag">Tag 1</Tag>
    <Tag href="#tag">Tag 2</Tag>
    <Tag href="#tag">Tag 3</Tag>
  </PinTags>
  <PinActions>
    <Action>edit</Action>
    <Action>like</Action>
    <Action variant="accent">later</Action>
    <Action variant="negative">archive</Action>
    <Spacer />
    <Link href="#status">
      <Status as="time" datetime="22:00" title="22:00">
        1 minute ago
      </Status>
    </Link>
  </PinActions>
</Pin>
```
